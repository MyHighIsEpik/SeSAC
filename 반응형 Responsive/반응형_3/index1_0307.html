@media all and (max-width:768px){ 사용자 해상도 768px 이하일 때 코드가 실행됨 }
@media all and (min-width:768px) and (max-width:1024px){ 사용자 해상도 768px 이상, 1024px 이하일 때 코드가 실행됨 }
@media all and (width:768px), (width:1024px){ 뷰포트의 너비가 768이거나 또는 1024일 때 }
@media all and (device-width:768px) and (device-width:1024px){ 스크린 너비가 768이고 그리고 1024일 때 }

@media all and (orientation:landscape){ 가로모드, 세로보다 가로가 길 때 }
